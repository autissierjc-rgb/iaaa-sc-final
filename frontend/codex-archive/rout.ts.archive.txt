function detectGate(text: string): GateMode {
  const t = ` ${text.toLowerCase()} `
  if (BLOCK_MARKERS.some((m) => t.includes(m))) return 'BLOCK'
  if (ANALYSIS_MARKERS.some((m) => t.includes(m))) return 'GENERATE'
  if (ORG_MARKERS.some((m) => t.includes(m))) return 'GENERATE'
  if (STATUS_MARKERS.some((m) => t.includes(m))) return 'FLASH'
  if (PERSONAL_MARKERS.some((m) => t.includes(m))) return 'CLARIFY'
  return 'GENERATE'
}

function extractTextContent(content: Array<{ type: string; text?: string }>): string {
  return content
    .filter((block) => block.type === 'text' && typeof block.text === 'string')
    .map((block) => block.text as string)
    .join('')
    .replace(/```json|```/g, '')
    .trim()
}

function parseJSON(raw: string): Record<string, unknown> {
  const clean = raw.replace(/```json|```/g, '').trim()

  const attempts = [
    clean,
    clean
      .replace(/[\u2018\u2019]/g, "'")
      .replace(/[\u201C\u201D]/g, '"')
      .replace(/\r?\n/g, ' '),
  ]

  for (const candidate of attempts) {
    try {
      return JSON.parse(candidate)
    } catch {}
  }

  const first = clean.indexOf('{')
  const last = clean.lastIndexOf('}')
  if (first !== -1 && last !== -1 && last > first) {
    const sliced = clean
      .slice(first, last + 1)
      .replace(/[\u2018\u2019]/g, "'")
      .replace(/[\u201C\u201D]/g, '"')
      .replace(/\r?\n/g, ' ')
    return JSON.parse(sliced)
  }

  throw new Error('Model returned invalid or truncated JSON')
}

function enrichWithScoring(
  sc: Record<string, unknown>,
  branches: AstrolabeBranch[]
): SituationCard {
  const radar = sc.radar as RadarScores | undefined
  if (!radar) return sc as SituationCard

  const state =
    branches.length > 0
      ? computeState(branches, radar)
      : Math.round(
          Math.max(
            0,
            Math.min(
              100,
              radar.impact * 0.3 +
                radar.urgency * 0.25 +
                radar.uncertainty * 0.25 +
                (100 - radar.reversibility) * 0.2
            )
          )
        )

  return {
    ...(sc as SituationCard),
    state_index_final: state,
    state_label: getStateLabel(state, 'fr'),
    state_label_en: getStateLabel(state, 'en'),
  }
}

async function generateFastCard(
  situation: string,
  branches: AstrolabeBranch[] = []
): Promise<SituationCard> {
  try {
    const msg = await client.messages.create({
      model: 'claude-sonnet-4-6',
      max_tokens: 1100,
      temperature: 0.2,
      messages: [
        {
          role: 'user',
          content: `${FAST_CARD_PROMPT}\n\nSituation:\n${situation}`,
        },
      ],
    })

    const raw = extractTextContent(msg.content as Array<{ type: string; text?: string }>)
    const sc = parseJSON(raw)
    return enrichWithScoring(sc, branches)
  } catch (err) {
    console.error('generateFastCard fallback:', err)
    return buildFallbackCard(situation, branches)
  }
}

async function generateFlash(text: string): Promise<FlashResponse | null> {
  try {
    const msg = await client.messages.create({
      model: 'claude-haiku-4-5-20251001',
      max_tokens: 220,
      messages: [
        {
          role: 'user',
          content: `${FLASH_PROMPT}\n\nQuestion: ${text}`,
        },
      ],
    })

    const raw = extractTextContent(msg.content as Array<{ type: string; text?: string }>)
    return JSON.parse(raw) as FlashResponse
  } catch {
    return null
  }
}

async function generateClarify(text: string): Promise<string[]> {
  try {
    const msg = await client.messages.create({
      model: 'claude-haiku-4-5-20251001',
      max_tokens: 120,
      messages: [
        {
          role: 'user',
          content: `${CLARIFY_PROMPT}\n\nSituation: ${text}`,
        },
      ],
    })

    const raw = extractTextContent(msg.content as Array<{ type: string; text?: string }>)
    const parsed = JSON.parse(raw) as ClarifyResponse
    return Array.isArray(parsed.questions) ? parsed.questions : []
  } catch {
    return []
  }
}

export async function POST(req: NextRequest) {
  try {
    const {
      situation,
      mode,
      lang = 'fr',
      astrolabe_branches,
    }: {
      situation?: string
      mode?: string
      lang?: 'fr' | 'en'
      astrolabe_branches?: AstrolabeBranch[]
    } = await req.json()

    if (!situation?.trim()) {
      return NextResponse.json({ error: 'No situation' }, { status: 400 })
    }

    const text = situation.trim()
    const branches = astrolabe_branches ?? []

    if (mode === 'generate' || mode === 'generate_full') {
      const sc = await generateFastCard(text, branches)
      const lecturePanel = await generateLecturePanel(text)

      const finalSc: SituationCard = {
        ...sc,
        lecture_systeme_fr:
          lecturePanel.lecture_systeme_fr || sc.lecture_systeme_fr,
        lecture_systeme_en:
          lecturePanel.lecture_systeme_en || sc.lecture_systeme_en,
      }

      return NextResponse.json({ gate: 'GENERATE', sc: finalSc })
    }

    const gate = detectGate(text)

    if (gate === 'BLOCK') {
      const reason =
        lang === 'fr'
          ? "Cette formulation ne permet pas d'analyse structurelle."
          : 'This formulation does not allow structural analysis.'
      return NextResponse.json({ gate: 'BLOCK', reason })
    }

    if (gate === 'FLASH') {
      const flash = await generateFlash(text)
      if (flash) {
        return NextResponse.json({ gate: 'FLASH', flash })
      }
      return NextResponse.json({ gate: 'GENERATE' })
    }

    if (gate === 'CLARIFY') {
      const questions = await generateClarify(text)
      return NextResponse.json({ gate: 'CLARIFY', questions })
    }

    return NextResponse.json({ gate: 'GENERATE' })
  } catch (err: unknown) {
    const error = err as {
      message?: string
      status?: number
      response?: unknown
      error?: unknown
    }

    console.error('generate error FULL:', err)
    console.error('generate error message:', error?.message)
    console.error('generate error status:', error?.status)
    console.error(
      'generate error response:',
      error?.response ?? error?.error ?? null
    )

    return NextResponse.json({ error: String(err) }, { status: 500 })
  }
}
